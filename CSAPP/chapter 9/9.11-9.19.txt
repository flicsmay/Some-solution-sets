 #include <stdio.h>
 #include <sys/types.h>
 #include <sys/stat.h>
 #include <fcntl.h>
 #include <unistd.h>
 #include <sys/mman.h>
 
 int main()
 {
	int fd;
	char *start;
	
	if ((fd = open("hello.txt", O_RDWR, 0)) < 0) { // in chapter 10
		printf("fail to open file\n");
		return 0;
	}
	
	start = mmap(NULL, 1, PROT_WRITE, MAP_SHARED, fd, 0);
	close(fd);
	
	if (start == MAP_FAILED) return -1;
	
	*start = 'J';
	munmap(start, 1);
	return 0;
 } // I have no idea how it work, but it actually does works.
